\begin{lstlisting}[language=Java, caption={2D Calculator (page.tsx)}]
"use client";
import { useRef, useState } from "react";
import Header from "../../components/Header";
import Footer from "../../components/Footer";
import "katex/dist/katex.min.css";

const fmt2 = (v: number): string => {
  if (Math.abs(v - Math.round(v)) < 1e-9) return Math.round(v).toString();
  return v.toFixed(2).replace(/\.?0+$/, "");
};

class ForceSystem2D {
  vectors: { fx: number; fy: number; magnitude: number; angleDeg: number }[];
  constructor() { this.vectors = []; }
  addForce(magnitude: number, angleDeg: number) {
    const angleRad = angleDeg * Math.PI / 180;
    const fx = magnitude * Math.cos(angleRad);
    const fy = magnitude * Math.sin(angleRad);
    this.vectors.push({ fx, fy, magnitude, angleDeg });
  }
  stepByStepSolution() {
    const steps: string[] = [];
    steps.push("Step 1: Resolve each force into components:");
    let sumFx = 0, sumFy = 0;
    this.vectors.forEach((v, i) => {
      steps.push(`Force ${i + 1}: |F|=${v.magnitude}, theta=${v.angleDeg}`);
      steps.push(`Fx${i+1} = ${fmt2(v.fx)}, Fy${i+1} = ${fmt2(v.fy)}`);
      sumFx += v.fx; sumFy += v.fy;
    });
    steps.push("Step 2: Sum of components:");
    steps.push(`SumFx = ${fmt2(sumFx)}, SumFy = ${fmt2(sumFy)}`);
    const R = Math.hypot(sumFx, sumFy);
    const theta = Math.atan2(sumFy, sumFx) * 180 / Math.PI;
    steps.push("Step 3: Resultant force:");
    steps.push(`R = ${R.toFixed(3)} kN, theta = ${theta.toFixed(2)} deg`);
    return { steps, sumFx, sumFy, R, theta };
  }
}

type ForceInput = { magnitude: string; angle: string };
type ForceResult = { steps: string[]; sumFx: number; sumFy: number; R: number; theta: number };

function ResultantFBD({ forces, result, svgRef }: {
  forces: ForceInput[]; result: ForceResult; svgRef?: React.RefObject<SVGSVGElement>
}) {
  const vectors = forces.map((f) => {
    const m = parseFloat(f.magnitude), a = parseFloat(f.angle);
    if (isNaN(m) || isNaN(a)) return null;
    const rad = a * Math.PI / 180;
    return { x: m * Math.cos(rad), y: m * Math.sin(rad) };
  }).filter(Boolean) as { x: number; y: number }[];
  const R = { x: result.sumFx, y: result.sumFy };
  const maxMag = Math.max(1, ...vectors.map(v => Math.hypot(v.x, v.y)),
    Math.hypot(R.x, R.y));
  const scale = 90 / maxMag;
  return (
    <svg ref={svgRef} width="300" height="300"
      className="border rounded-lg bg-white shadow mx-auto">
      <g transform="translate(150,150)">
        <line x1={-140} y1={0} x2={140} y2={0} stroke="gray"/>
        <line x1={0} y1={-140} x2={0} y2={140} stroke="gray"/>
        {vectors.map((v, i) => {
          const x = v.x * scale, y = -v.y * scale;
          return (
            <g key={i}>
              <line x1={0} y1={0} x2={x} y2={y}
                stroke="#1848a0" strokeWidth="3" markerEnd="url(#arrowF)"/>
              <text x={x+6} y={y-6} fontSize="14" fill="#1848a0">F{i+1}</text>
            </g>
          );
        })}
        <line x1={0} y1={0} x2={R.x*scale} y2={-R.y*scale}
          stroke="#009900" strokeWidth="4" markerEnd="url(#arrowR)"/>
        <text x={R.x*scale+8} y={-R.y*scale-8} fontSize="16" fill="#009900">R</text>
      </g>
    </svg>
  );
}

export default function Solver2D() {
  const [forces, setForces] = useState([{ magnitude: "", angle: "" }]);

class ForceSystem3D {
  vectors: any[] = [];
  addForce(magnitude: number, azimuthDeg: number, elevationDeg: number) {
    const az = azimuthDeg * Math.PI / 180;
    const el = elevationDeg * Math.PI / 180;
    this.vectors.push({
      fx: magnitude * Math.cos(el) * Math.cos(az),
      fy: magnitude * Math.cos(el) * Math.sin(az),
      fz: magnitude * Math.sin(el),
      magnitude, azimuthDeg, elevationDeg
    });
  }
  solve() {
    const steps: string[] = [];
    steps.push("Step 1: Resolve each force into 3D components:");
    let sumFx = 0, sumFy = 0, sumFz = 0;
    this.vectors.forEach((v, i) => {
      steps.push(`Force ${i+1}: |F|=${v.magnitude}, phi=${v.azimuthDeg}, alpha=${v.elevationDeg}`);
      steps.push(`Fx=${v.fx.toFixed(3)}, Fy=${v.fy.toFixed(3)}, Fz=${v.fz.toFixed(3)}`);
      sumFx += v.fx; sumFy += v.fy; sumFz += v.fz;
    });
    steps.push("Step 2: Sum of components:");
    steps.push(`SumFx=${sumFx.toFixed(3)}, SumFy=${sumFy.toFixed(3)}, SumFz=${sumFz.toFixed(3)}`);
    const R = Math.sqrt(sumFx**2 + sumFy**2 + sumFz**2);
    const azimuth = Math.atan2(sumFy, sumFx) * 180 / Math.PI;
    const elevation = Math.asin(sumFz / (R || 1)) * 180 / Math.PI;
    steps.push("Step 3: Resultant force:");
    steps.push(`R=${R.toFixed(3)} kN, phi=${azimuth.toFixed(2)}, alpha=${elevation.toFixed(2)}`);
    if (R > 0.0001) {
      const alpha = Math.acos(sumFx / R) * 180 / Math.PI;
      const beta  = Math.acos(sumFy / R) * 180 / Math.PI;
      const gamma = Math.acos(sumFz / R) * 180 / Math.PI;
      steps.push("Step 4: Direction angles:");
      steps.push(`alpha=${alpha.toFixed(2)}, beta=${beta.toFixed(2)}, gamma=${gamma.toFixed(2)}`);
    }
    return { steps, sumFx, sumFy, sumFz, R, azimuth, elevation };
  }
}

function buildSolution(details: any[], Rx: number, Ry: number, Rz: number, R: number) {
  const steps = [];
  steps.push("Step 1: Determine position vectors");
  details.forEach(d =>
    steps.push(`r_${d.from}${d.to} = (${d.dx})i + (${d.dy})j + (${d.dz})k`));

function calculate(forces: ForceInput[]) {
  const magUnknownIndices = forces.map((f, i) => f.magnitudeUnknown ? i : -1).filter(i => i !== -1);
  const totalUnknowns = magUnknownIndices.length;
  if (totalUnknowns > 2) return { error: "Too many unknowns" };
  const knownForces = forces
    .map((f, i) => ({ f, i }))
    .filter(({ f, i }) => !magUnknownIndices.includes(i))
    .map(({ f, i }) => {
      const magnitude = parseFloat(f.magnitude);
      const angle = parseFloat(f.angle);
      if (isNaN(magnitude) || isNaN(angle)) return null;
      return { magnitude, angle, label: `F_${i+1}` };

    const Rx = -resultX.unknowns[0].value;
    const Ry = -resultX.unknowns[1].value;

}

function KTX({ tex }: { tex: string }) {
  const ref = useCallback((el: HTMLDivElement | null) => {
    if (!el || !window.katex) return;
    try { window.katex.render(tex, el, { displayMode: true, throwOnError: false }); }
    catch { el.innerText = tex; }
  }, [tex]);
  const ready = useKatexScript();
  if (!ready) return null;
  return <div ref={ref}/>;
}
